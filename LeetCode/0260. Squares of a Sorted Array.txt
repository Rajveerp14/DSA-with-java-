class Solution {
public int[] sortedSquares(int[] nums) {
int n = nums.length;
int[] ans = new int[n];
int left = 0;
int right = n - 1;
int i = n - 1;
while (left <= right) {
if (Math.abs(nums[left]) > Math.abs(nums[right])) {
ans[i] = nums[left] * nums[left];
} else {
ans[i] = nums[right] * nums[right];
}
}
return ans;
}
i--;
left++;
right--;
}
